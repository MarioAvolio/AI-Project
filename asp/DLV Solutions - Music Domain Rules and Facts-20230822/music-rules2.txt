% band and members from member_of
music_work(X) :- artist_of(_,X).

% infer genre(X) from relations  
genre(X) :- subgenre(X,_).
genre(X) :- subgenre(_,X).
genre(X) :- music_genre(_,X).

% inheritance with unary predicates   
artist(X) :- musician(X).
artist(X) :- singer(X).
artist(X) :- band(X).
